264  void CollisionBall(poolBall *A, poolBall *B)
265  //the given code coordinate of A and B, velocity of A and B, mass of A and B.
266  {
267  //Properties of two colliding balls,
268 
269  QVector2D posA(A->getcoordinate().getQtRenderingXCoordinate(), A->getcoordinate().getQtRenderingYCoordinate());
270  QVector2D velA(A->getvelocity().getXVelocity(), A->getvelocity().getYVelocity());
271  float massA = A->getmass();
272  //and ball B
273  QVector2D posB(B->getcoordinate().getQtRenderingXCoordinate(), B->getcoordinate().getQtRenderingYCoordinate());
274  QVector2D velB(B->getvelocity().getXVelocity(), B->getvelocity().getYVelocity());
275  float massB = B->getmass();
276 
277  //calculate their mass ratio
278  float mR = massB / massA;
279 
280  //calculate the axis of collision
281  QVector2D collisionVector = posB - posA;
282  collisionVector.normalize();
283 
284  //the proportion of each balls velocity along the axis of collision
285  double vA = QVector2D::dotProduct(collisionVector, velA);
286  double vB = QVector2D::dotProduct(collisionVector, velB);
287  //the balls are moving away from each other so do nothing
288  if (vA <= 0 && vB >= 0) {
289  return ;
290  }
291 
292  //The velocity of each ball after a collision can be found by solving the quadratic equation
293  //given by equating momentum and energy before and after the collision and finding the velocities
294  //that satisfy this
295  //-(mR+1)x^2 2*(mR*vB+vA)x -((mR-1)*vB^2+2*vA*vB)=0
296  //first we find the discriminant
297  double a = -(mR + 1);
298  double b = 2 * (mR * vB + vA);
299  double c = -((mR - 1) * vB * vB + 2 * vA * vB);
300  double discriminant = sqrt(b * b - 4 * a * c);
301  double root = (-b + discriminant)/(2 * a);
302  //only one of the roots is the solution, the other pertains to the current velocities
303  if (root - vB < 0.01) {
304  root = (-b - discriminant)/(2 * a);
305  }
306 
307 
308  //The resulting changes in velocity for ball A and B
309  QVector2D deltaVA = mR * (vB - root) * collisionVector;
310  QVector2D deltaVB = (root - vB) * collisionVector;
311 
312  float x1 = deltaVA.x();
313  float y1 = deltaVA.y();
314  float x2 = deltaVB.x();
315  float y2 = deltaVB.y();
316 
317  A->changeVX(x1);
318  A->changeVY(y1);
319  B->changeVX(x2);
320  B->changeVY(y2);
321  }
